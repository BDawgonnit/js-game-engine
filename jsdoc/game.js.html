/**
 * Game module.
 * @module game
 */
define('Game', function(module) {
	'use strict';

	const EntityFactory = require('EntityFactory'); // TODO: Move to GameEntityFactory module with code below
	const Engine = require('Engine');
	const RenderSystem = require('RenderSystem');

	// TODO: Move to its own module
	// TODO: Clean up - Make sure Tiled-specific data structure isn't a dependency (don't use tiled JSON directly)
	class GameEntityFactory extends EntityFactory {
		create(entityType, data, compAddCallback, compDelCallback) {
			let entity = super.create(compAddCallback, compDelCallback);
			switch(entityType) {
				case 'Tunnel': entity.addComponent('tunnelDetails', { sendTo: data.properties.sendTo }); break;
				case 'AreaGate': entity.addComponent('gate', {x: 1, y: 1}); break;
			}
			return entity;
		}
	}

	/** Class representing a Game. */
	class Game {

		/**
		 * Create a Game instance.
		 * @param  {string} jsonPath - File path to map .json file.
		 */
		constructor() {
			this.engine = new Engine('json/level2.json', new GameEntityFactory(), () => {
				this.engine.addSystem('render', new RenderSystem(this.engine.map));
			});
		}

		run() {
			this.engine.run();
		}
	}

	module.exports = Game;
});
